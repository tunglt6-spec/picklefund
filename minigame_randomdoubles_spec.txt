2.2 Flow chi tiết – Bốc thăm lượt
Club Admin nhấn [Bốc Thăm Lượt Mới]
  ↓ Chọn draw_mode: RANDOM | FAIR_ROTATION | BALANCED_SKILL
  ↓ POST /minigames/:id/draw-round
  ↓ Backend:
       1. Tổng hợp participants (status=ACTIVE)
       2. Sắp xếp theo draw_mode (random / fair / balanced)
       3. Lấy floor(n/4)*4 người → tạo matches
       4. Phần dư → sit_out (ghi vào MiniGameRoundSitOuts)
       5. INSERT MiniGameRound + MiniGameMatches + MiniGameMatchPlayers
  ↓ Response: round info + matches[team1[2 players], team2[2 players]] + sitOuts[]
  ↓ UI hiển thị Match Cards 2 vs 2

2.3 Flow nhập kết quả
Admin nhấn [Nhập KQ] trên Match Card
  ↓ Drawer mở: Team 1 [ -  11  + ]  vs  Team 2 [ -  7  + ]
  ↓ PUT /minigames/:id/matches/:matchId/result { team1Score:11, team2Score:7 }
  ↓ Backend transaction:
       1. Xác định winning_team (team với score cao hơn)
       2. UPDATE MiniGameMatches (team1_score, team2_score, winning_team, status=COMPLETED)
       3. UPDATE MiniGameMatchPlayers (is_winner, score_for, score_against, ranking_points)
       4. UPSERT MiniGameStandings x4 người (played+1, won/lost+1, points,...)
       5. UPSERT MiniGamePairStats (Team1 pair & Team2 pair)
       6. UPSERT MiniGameOpponentStats (Team1[0] vs Team2[0,1], Team1[1] vs Team2[0,1])
       7. INSERT MiniGameAuditLog
  ↓ Response: { result, standings_delta }

4.1 Thuật toán bốc thăm – RANDOM
// minigame.service.ts – drawRound (RANDOM)
function drawRandom(participants: Member[]): DrawResult {
  const shuffled = [...participants].sort(() => Math.random() - 0.5);
  const matchCount = Math.floor(shuffled.length / 4);
  const playing = shuffled.slice(0, matchCount * 4);
  const sitOuts = shuffled.slice(matchCount * 4);

  const matches = [];
  for (let i = 0; i < matchCount; i++) {
    const group = playing.slice(i * 4, i * 4 + 4);
    // random chia team: [0,1] vs [2,3] hoặc [0,2] vs [1,3] hoặc [0,3] vs [1,2]
    const splits = [[0,1,2,3],[0,2,1,3],[0,3,1,2]];
    const s = splits[Math.floor(Math.random() * 3)];
    matches.push({
      team1: [group[s[0]], group[s[1]]],
      team2: [group[s[2]], group[s[3]]],
    });
  }
  return { matches, sitOuts };
}

4.2 Thuật toán bốc thăm – FAIR_ROTATION
// Sắp xếp theo công bằng: ít trận → nhiều ngồi ngoài → lâu chơi nhất
function fairSort(participants: ParticipantWithStats[]): Member[] {
  return [...participants].sort((a, b) => {
    if (a.played !== b.played) return a.played - b.played;        // ít trận hơn lên trước
    if (a.sitOutCount !== b.sitOutCount) return b.sitOutCount - a.sitOutCount; // ngồi ngoài nhiều lên trước
    return a.lastPlayedAt - b.lastPlayedAt;                        // lâu chơi nhất lên trước
  });
}

// Chọn phương án chia team tốt nhất trong nhóm 4
function bestSplit(group: Member[], pairHistory: PairStats[]): [Member[], Member[]] {
  const splits = [
    [[0,1],[2,3]], [[0,2],[1,3]], [[0,3],[1,2]]
  ];
  let best = null; let bestScore = Infinity;
  for (const [t1idx, t2idx] of splits) {
    const t1 = t1idx.map(i => group[i]);
    const t2 = t2idx.map(i => group[i]);
    const score = pairRepeatPenalty(t1, t2, pairHistory)
                + opponentRepeatPenalty(t1, t2)
                + skillImbalancePenalty(t1, t2);
    if (score < bestScore) { bestScore = score; best = [t1, t2]; }
  }
  return best!;
}

// Penalty: cặp đã chơi cùng nhiều lần
function pairRepeatPenalty(t1, t2, history) {
  const getPaired = (a, b) => history.find(h => 
    (h.memberAId === a.id && h.memberBId === b.id) ||
    (h.memberAId === b.id && h.memberBId === a.id)
  )?.pairedCount ?? 0;
  return getPaired(t1[0], t1[1]) + getPaired(t2[0], t2[1]);
}

4.3 Thuật toán BALANCED_SKILL
// Cân bằng trình độ: |sum_skill(team1) - sum_skill(team2)| tối thiểu
function balancedSplit(group: Member[]): [Member[], Member[]] {
  const splits = [[[0,1],[2,3]], [[0,2],[1,3]], [[0,3],[1,2]]];
  let best = splits[0]; let minDiff = Infinity;
  for (const [t1idx, t2idx] of splits) {
    const s1 = t1idx.reduce((s,i) => s + (group[i].skillLevel ?? 50), 0);
    const s2 = t2idx.reduce((s,i) => s + (group[i].skillLevel ?? 50), 0);
    const diff = Math.abs(s1 - s2);
    if (diff < minDiff) { minDiff = diff; best = [t1idx, t2idx]; }
  }
  return [best[0].map(i=>group[i]), best[1].map(i=>group[i])];
}

4.4 Thuật toán tính điểm cá nhân (sau nhập kết quả)
// Sau PUT /matches/:id/result – chạy trong DB transaction
async function updateStandingsAfterMatch(
  matchId: string, team1Score: number, team2Score: number, ctx: PrismaClient
) {
  const match = await ctx.miniGameMatch.findUnique({ where: { id: matchId },
    include: { players: true, miniGame: true } });

  const winningTeam = team1Score > team2Score ? 1 : team2Score > team1Score ? 2 : 0; // 0=hòa
  const { winPoints, drawPoints, lossPoints, allowDraw } = match.miniGame;

  if (winningTeam === 0 && !allowDraw)
    throw new BadRequestException("Minigame này không cho phép kết quả hòa");

  const updates = match.players.map(p => {
    const myScore   = p.teamNumber === 1 ? team1Score : team2Score;
    const oppScore  = p.teamNumber === 1 ? team2Score : team1Score;
    const isWinner  = winningTeam === p.teamNumber;
    const isDraw    = winningTeam === 0;
    const pts       = isWinner ? winPoints : isDraw ? drawPoints : lossPoints;

    return ctx.miniGameStanding.upsert({
      where:  { minigameId_memberId: { minigameId: match.minigameId, memberId: p.memberId } },
      create: { clubId: match.clubId, minigameId: match.minigameId, memberId: p.memberId,
                played:1, won: isWinner?1:0, drawn: isDraw?1:0, lost: !isWinner&&!isDraw?1:0,
                pointsFor: myScore, pointsAgainst: oppScore, rankingPoints: pts },
      update: { played: { increment:1 }, won: { increment: isWinner?1:0 },
                drawn: { increment: isDraw?1:0 }, lost: { increment: !isWinner&&!isDraw?1:0 },
                pointsFor: { increment: myScore }, pointsAgainst: { increment: oppScore },
                rankingPoints: { increment: pts } }
    });
  });

  await ctx.$transaction(updates);
  await recalcWinRate(match.minigameId, ctx);
  await recalcPairStats(match, winningTeam, ctx);
  await recalcOpponentStats(match, winningTeam, ctx);
}

// components/MatchCard2v2.tsx
interface Match {
  id: string; matchNumber: number; status: string;
  team1: Player[]; team2: Player[];
  team1Score?: number; team2Score?: number; winningTeam?: number;
}

export function MatchCard2v2({ match, onEnterScore }: { match: Match; onEnterScore: () => void }) {
  const isCompleted = match.status === "COMPLETED";
  const team1Won = match.winningTeam === 1;
  const team2Won = match.winningTeam === 2;

  return (
    <div className="bg-white rounded-2xl shadow-sm border border-gray-100 p-4">
      <div className="flex items-center justify-between mb-3">
        <span className="font-semibold text-gray-700">Trận {match.matchNumber}</span>
        <StatusBadge status={match.status} />
      </div>
      <div className="grid grid-cols-[1fr_auto_1fr] gap-3 items-center">
        {/* Team 1 */}
        <div className={`rounded-xl p-3 ${team1Won ? "bg-green-50 border-2 border-green-400" : "bg-gray-50"}`}>
          {team1Won && <span className="text-xs font-bold text-green-600">🏆 THẮNG</span>}
          {match.team1.map(p => <p key={p.id} className="font-medium text-sm">{p.name}</p>)}
        </div>
        {/* Score */}
        <div className="text-center">
          {isCompleted ? (
            <div className="text-2xl font-bold text-gray-800">
              {match.team1Score} – {match.team2Score}
            </div>
          ) : <span className="text-gray-400 font-semibold">VS</span>}
        </div>
        {/* Team 2 */}
        <div className={`rounded-xl p-3 ${team2Won ? "bg-green-50 border-2 border-green-400" : "bg-gray-50"}`}>
          {team2Won && <span className="text-xs font-bold text-green-600">🏆 THẮNG</span>}
          {match.team2.map(p => <p key={p.id} className="font-medium text-sm">{p.name}</p>)}
        </div>
      </div>
      {!isCompleted && (
        <button onClick={onEnterScore}
          className="mt-3 w-full py-2 bg-indigo-600 text-white rounded-lg text-sm font-medium">
          ✏ Nhập Kết Quả
        </button>
      )}
    </div>
  );
}

9.3 Multi-tenant Guard
@Injectable()
export class MinigameGuard implements CanActivate {
  async canActivate(ctx: ExecutionContext): Promise<boolean> {
    const req = ctx.switchToHttp().getRequest();
    const { clubId, role } = req.user; // từ JWT - KHÔNG từ client
    if (role === "SUPER_ADMIN") return true;
    const minigame = await this.prisma.miniGame.findFirst({
      where: { id: req.params.id, clubId }  // BẮT BUỘC: clubId từ token
    });
    if (!minigame) throw new ForbiddenException("Access denied");
    req.minigame = minigame; // cache
    return true;
  }
}

    expect(standingA.pointsFor).toBe(11);
    expect(standingA.pointsAgainst).toBe(7);
  });

  it("team2 players get 0 ranking points when losing", async () => {
    await service.processMatchResult(matchId, 11, 7);
    const standingC = await getStanding(matchId, "playerC");
    expect(standingC.lost).toBe(1);
    expect(standingC.rankingPoints).toBe(0);
  });

  it("throws error when draw and allow_draw=false", async () => {
    await expect(service.processMatchResult(matchId, 7, 7))
      .rejects.toThrow("Draw not allowed");
  });

  it("win_rate recalculated after each match", async () => {
    await service.processMatchResult(m1, 11, 7); // win
    await service.processMatchResult(m2, 5, 11); // loss
    const s = await getStanding(minigameId, "playerA");
    expect(s.winRate).toBe(50.00);
  });

  it("rollback standings when result is deleted", async () => {
    await service.processMatchResult(matchId, 11, 7);
    const before = await getStanding(minigameId, "playerA");
    await service.deleteMatchResult(matchId);
    const after = await getStanding(minigameId, "playerA");
    expect(after.played).toBe(before.played - 1);
    expect(after.rankingPoints).toBe(before.rankingPoints - 3);
  });
});
